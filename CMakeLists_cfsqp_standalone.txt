PROJECT(CFSQP)

CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

##
## Version stuff
##
SET(OPENSIM_CFSQP_LIBRARY_NAME osimCFSQP)
SET(OPENSIM_CFSQP_MAJOR_VERSION 3)
SET(OPENSIM_CFSQP_MINOR_VERSION 0)
SET(OPENSIM_CFSQP_BUILD_VERSION 0)
SET(OPENSIM_CFSQP_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_CFSQP_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_CFSQP_LIBRARY_NAME=${OPENSIM_CFSQP_LIBRARY_NAME}
-DOPENSIM_CFSQP_MAJOR_VERSION=${OPENSIM_CFSQP_MAJOR_VERSION}
-DOPENSIM_CFSQP_MINOR_VERSION=${OPENSIM_CFSQP_MINOR_VERSION}
-DOPENSIM_CFSQP_BUILD_VERSION=${OPENSIM_CFSQP_BUILD_VERSION}
-DOPENSIM_CFSQP_COPYRIGHT_YEARS="${OPENSIM_CFSQP_COPYRIGHT_YEARS}"
-DOPENSIM_CFSQP_AUTHORS="${OPENSIM_CFSQP_AUTHORS}"
-DOPENSIM_CFSQP_TYPE="Shared")

SET(SOURCE_FILES cfsqp.c qld.c osimCFSQPDLL.cpp About.cpp)
FILE(GLOB INCLUDE_FILES *.h)

IF(WIN32)
    ADD_DEFINITIONS("-DOSIMCFSQP_EXPORTS -D__STDC__")
ELSE(WIN32)
    #ADD_DEFINITIONS("-DOSIMCFSQP_EXPORTS")
ENDIF(WIN32)

ADD_LIBRARY(osimCFSQP SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 

SET_TARGET_PROPERTIES(osimCFSQP
    PROPERTIES PROJECT_LABEL "Vendor Libraries - osimCFSQP")

INSTALL(TARGETS osimCFSQP
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION sdk/lib)
INSTALL(FILES ${INCLUDE_FILES} DESTINATION include/Vendors/CFSQP)
